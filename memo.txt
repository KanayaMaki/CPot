Game
	virtual void Setting()
		ゲームが立ち上がってすぐの段階で、画面サイズやそのほかの設定をする
	virtual void Init()
		ゲームの初期化
	virtual void Update()
		ゲームの更新をする
	virtual void Render()
		ゲームの描画をする

class GameMain
	static GameBase* mGame

	static Main() {
		//初期化
		mGame->Init();

		while(true) {
			Input->Update();
			mGame->Update();
			mGame->Render();
			Fps->Wait();

			if(mGameEnd) {
				break;
			}
		}
	}
